# Cart Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement a fully functional shopping cart with a slide-in drawer, add-to-cart from the product page, React 19 `useOptimistic` quantity editing, and a standalone cart page.

**Architecture:** `CartProvider` (client context) holds `cart` state and drawer open/close state; it receives the initial cart fetched server-side in the locale layout. All Shopify mutations go through Server Actions in `lib/cart.ts` which always return the updated `ShopifyCart`. `CartLineItem` uses `useOptimistic` + `useTransition` so quantity changes appear instant. The `CartDrawer` is rendered once in the layout, inside `CartProvider`, so it is always available in the component tree.

**Tech Stack:** Next.js 16 App Router, React 19 (`useOptimistic`, `useTransition`), next-intl v4, Tailwind CSS v4 semantic tokens, Vitest + @testing-library/react + @testing-library/user-event

// providers/CartProvider.tsx
'use client'

import { createContext, useContext, useState, useCallback } from 'react'
import type { ShopifyCart } from '@/lib/shopify/types'

type CartContextValue = {
  cart: ShopifyCart | null
  isOpen: boolean
  openCart: () => void
  closeCart: () => void
  setCart: (cart: ShopifyCart) => void
}

const CartContext = createContext<CartContextValue | null>(null)

export function useCart(): CartContextValue {
  const ctx = useContext(CartContext)
  if (!ctx) throw new Error('useCart must be used inside CartProvider')
  return ctx
}

type Props = {
  initialCart: ShopifyCart | null
  children: React.ReactNode
}

export function CartProvider({ initialCart, children }: Props) {
  const [cart, setCartState] = useState<ShopifyCart | null>(initialCart)
  const [isOpen, setIsOpen] = useState(false)

  const openCart = useCallback(() => setIsOpen(true), [])
  const closeCart = useCallback(() => setIsOpen(false), [])
  const setCart = useCallback((c: ShopifyCart) => setCartState(c), [])

  return (
    <CartContext.Provider value={{ cart, isOpen, openCart, closeCart, setCart }}>
      {children}
    </CartContext.Provider>
  )
}
```

// components/store/CartLineItem.tsx
'use client'

import { useOptimistic, useTransition } from 'react'
import Image from 'next/image'
import { useTranslations } from 'next-intl'
import type { CartLine } from '@/lib/shopify/types'

type Props = {
  line: CartLine
  onUpdate: (lineId: string, quantity: number) => Promise<void>
  onRemove: (lineId: string) => Promise<void>
}

export function CartLineItem({ line, onUpdate, onRemove }: Props) {
  const t = useTranslations('cart')
  const [isPending, startTransition] = useTransition()
  const [optimisticQty, addOptimistic] = useOptimistic(
    line.quantity,
    (_: number, next: number) => next
  )

  const image = line.merchandise.product.featuredImage

  function handleDecrease() {
    if (optimisticQty <= 1) {
      startTransition(async () => {
        addOptimistic(0)
        await onRemove(line.id)
      })
    } else {
      const next = optimisticQty - 1
      startTransition(async () => {
        addOptimistic(next)
        await onUpdate(line.id, next)
      })
    }
  }

  function handleIncrease() {
    const next = optimisticQty + 1
    startTransition(async () => {
      addOptimistic(next)
      await onUpdate(line.id, next)
    })
  }

  function handleRemove() {
    startTransition(async () => {
      addOptimistic(0)
      await onRemove(line.id)
    })
  }

  return (
    <div className={`flex gap-3 py-4 border-b border-border last:border-b-0 transition-opacity ${isPending ? 'opacity-60' : 'opacity-100'}`}>
      <div className="w-16 h-16 flex-shrink-0 rounded-md overflow-hidden bg-muted">
        {image ? (
          <Image
            src={image.url}
            alt={image.altText ?? line.merchandise.product.title}
            width={64}
            height={64}
            className="w-full h-full object-cover"
          />
        ) : (
          <div className="w-full h-full flex items-center justify-center text-muted-foreground text-xs">
            —
          </div>
        )}
      </div>
      <div className="flex-1 min-w-0">
        <p className="text-sm font-medium text-foreground truncate">
          {line.merchandise.product.title}
        </p>
        <p className="text-xs text-muted-foreground mt-0.5">{line.merchandise.title}</p>
        <p className="text-sm font-medium text-foreground mt-1">
          {line.merchandise.price.currencyCode}{' '}
          {parseFloat(line.merchandise.price.amount).toFixed(2)}
        </p>
        <div className="flex items-center gap-2 mt-2">
          <button
            aria-label="Decrease quantity"
            onClick={handleDecrease}
            disabled={isPending}
            className="w-6 h-6 flex items-center justify-center border border-border rounded text-foreground hover:bg-muted transition-colors text-sm disabled:opacity-40"
          >
            −
          </button>
          <span className="text-sm w-4 text-center text-foreground">{optimisticQty}</span>
          <button
            aria-label="Increase quantity"
            onClick={handleIncrease}
            disabled={isPending}
            className="w-6 h-6 flex items-center justify-center border border-border rounded text-foreground hover:bg-muted transition-colors text-sm disabled:opacity-40"
          >
            +
          </button>
          <button
            aria-label="Remove item"
            onClick={handleRemove}
            disabled={isPending}
            className="ml-auto text-xs text-muted-foreground hover:text-foreground transition-colors disabled:opacity-40"
          >
            {t('remove')}
          </button>
        </div>
      </div>
    </div>
  )
}
```

// components/store/CartDrawer.tsx
'use client'

import Link from 'next/link'
import { useTranslations } from 'next-intl'
import { useCart } from '@/providers/CartProvider'
import { CartLineItem } from '@/components/store/CartLineItem'
import { updateCartAction, removeFromCartAction } from '@/lib/cart'

export function CartDrawer() {
  const tCart = useTranslations('cart')
  const tCommon = useTranslations('common')
  const { cart, isOpen, closeCart, setCart } = useCart()
  const lines = cart?.lines.nodes ?? []

  async function handleUpdate(lineId: string, quantity: number) {
    const updated = await updateCartAction(lineId, quantity)
    setCart(updated)
  }

  async function handleRemove(lineId: string) {
    const updated = await removeFromCartAction(lineId)
    setCart(updated)
  }

  return (
    <>
      <div
        className={`fixed inset-0 z-40 bg-foreground/40 transition-opacity duration-300 ${
          isOpen ? 'opacity-100' : 'opacity-0 pointer-events-none'
        }`}
        onClick={closeCart}
        aria-hidden="true"
      />
      <div
        role="dialog"
        aria-modal="true"
        aria-label={tCart('title')}
        className={`fixed right-0 top-0 bottom-0 z-50 w-full max-w-md bg-background flex flex-col shadow-lg transition-transform duration-300 ${
          isOpen ? 'translate-x-0' : 'translate-x-full'
        }`}
      >
        <div className="flex items-center justify-between px-4 py-3 border-b border-border">
          <h2 className="text-lg font-semibold text-foreground">{tCart('title')}</h2>
          <button
            onClick={closeCart}
            aria-label={tCommon('close')}
            className="w-8 h-8 flex items-center justify-center rounded text-muted-foreground hover:text-foreground hover:bg-muted transition-colors"
          >
            ✕
          </button>
        </div>
        <div className="flex-1 overflow-y-auto px-4">
          {lines.length === 0 ? (
            <p className="text-muted-foreground text-sm mt-8 text-center">{tCart('empty')}</p>
          ) : (
            lines.map((line) => (
              <CartLineItem
                key={line.id}
                line={line}
                onUpdate={handleUpdate}
                onRemove={handleRemove}
              />
            ))
          )}
        </div>
        {cart && lines.length > 0 && (
          <div className="border-t border-border px-4 py-4 space-y-3">
            <div className="flex justify-between text-sm">
              <span className="text-muted-foreground">{tCart('subtotal')}</span>
              <span className="font-medium text-foreground">
                {cart.cost.subtotalAmount.currencyCode}{' '}
                {parseFloat(cart.cost.subtotalAmount.amount).toFixed(2)}
              </span>
            </div>
            <Link
              href={cart.checkoutUrl}
              className="block w-full text-center py-3 bg-primary text-primary-foreground font-medium rounded-lg hover:bg-primary/90 transition-colors"
            >
              {tCart('checkout')}
            </Link>

// components/store/ProductForm.tsx
'use client'

import { useState, useTransition } from 'react'
import { useTranslations } from 'next-intl'
import { VariantSelector } from './VariantSelector'
import { useCart } from '@/providers/CartProvider'
import { addToCartAction } from '@/lib/cart'
import type { ShopifyProduct, ProductVariant } from '@/lib/shopify/types'

type Props = {
  product: ShopifyProduct
}

export function ProductForm({ product }: Props) {
  const t = useTranslations('product')
  const { setCart, openCart } = useCart()
  const [isPending, startTransition] = useTransition()

  const isSingleVariant =
    product.options.length === 1 &&
    product.options[0].name === 'Title' &&
    product.options[0].values[0] === 'Default Title'

  const [selectedVariant, setSelectedVariant] = useState<ProductVariant | null>(
    product.variants.nodes[0] ?? null
  )

  const available = isSingleVariant
    ? product.availableForSale
    : (selectedVariant?.availableForSale ?? false)

  function handleAddToCart() {
    if (!selectedVariant) return
    startTransition(async () => {
      const updated = await addToCartAction(selectedVariant.id)
      setCart(updated)
      openCart()
    })
  }

  return (
    <div className="space-y-4">
      <VariantSelector
        options={product.options}
        variants={product.variants.nodes}
        onVariantChange={setSelectedVariant}
      />
      <button
        disabled={!available || isPending}
        onClick={handleAddToCart}
        className="w-full py-3 px-6 bg-primary text-primary-foreground font-medium rounded-lg disabled:bg-muted disabled:text-muted-foreground disabled:cursor-not-allowed hover:bg-primary/80 transition-colors"
      >
        {isPending ? t('addingToCart') : available ? t('addToCart') : t('outOfStock')}
      </button>
    </div>
  )
}
